- 实现模板匹配
```js
function template(pattern) {
  return function(str) {
    var res = '';
    for (var key in str) {
      var regexp = new RegExp('<%= '+ key +' %>')
      res = pattern.replace(regexp, str[key])
    }
    return res
  }
}
var compiled = template('hello <%= user %>!');
compiled({'user': 'world'})
```
